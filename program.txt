	  START 101
	  READ N
	  MOVER BREG, ='1'
	  MOVEM BREG, TERM
AGAIN MULT BREG, TERM
	  MOVER CREG, TERM
	  ADD CREG, ONE
	  MOVEM CREG, N
		COMP CREG, N
	  BC LE, AGAIN
	  LTORG
	  DIV BREG, ='2'
READ BREG, ='4'
	  MOVEM BREG, RESULT
	  PRINT RESULT
	  STOP
N 	  DS 1
RESULT DS 1
ONE   DC '1'
TERM  DS  1
		END
